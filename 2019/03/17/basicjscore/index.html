//也可定义自己的方法：this关键字是对定义方法的对象的引用
points.dist = function(){ //定义一个方法用来计算两点之间的距离
var p1 = this[0]; //通过this获得对当前数组的引用
var p2 = this[1]; //并取得调用数组的前两个元素
var a = p2.x - p1.x; //x坐标轴上的距离
var b = p2.y - p1.y; //y坐标轴上的距离
return Math.sqrt(aa + bb); //用Math.sqrt()来计算平方根
};

function factorial(n){ //计算阶乘的函数
var product = 1;
while (n > 1){ //当()内的表达式为true时循环执行{}内的代码
product = n; // product = product n;
n–; // n = n - 1;
} //循环结束
return product;
}
factorial(4) // => 24
function factorial2(n){ //实现循环的另一种写法
var i,product =1;
for(i = 2;i <= n;i++) //将i从2自增到n
product *= 1; //循环体，当循环体中只有一句代码，可以省略{}
return product; //返回计算好的阶乘
factorial2(5) // => 120
}

} //不需要return

//使用new关键字和构造函数来创建一个实例

//通过给构造函数的prototye对象赋值来给Point对象定义方法
Point.prototype.r = function(){
return Math.sqrt(this.xthis.x + this.ythis.y); //this指代调用这个方法的对象
};

//Point的实例对象p(以及所有的Point实例对象)继承了方法r()
p.r() // => 1.414…
